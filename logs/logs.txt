    X: Any = dataset['X']
    y: Any = dataset['y']
    learning_rate: str = cfg.get('learning_rate')
    alpha: float = cfg.get('alpha')
    max_iter: int = cfg.get('max_iter')
    eta0: float = cfg.get('eta0')

    C: float = 1.0 / alpha
    params: dict = {
        'loss': 'log_loss',
